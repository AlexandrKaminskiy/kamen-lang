procedure entrypoint()
begin
    var context: Context
    var circle: Shape
    context := CreateContext("out.svg", 800, 600)
    for i in 1 to 9
    begin
        #circle := DrawCircle(50.0 * i, 100.0, 20)
        AddToContext(context, DrawCircle(50.0 * i, 100.0, 20))
    end

    Render(context)
end
