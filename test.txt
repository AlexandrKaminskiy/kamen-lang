function fibonacci(var n: Integer): Integer
begin
    var result: Integer

    if (n <= 1)
    begin
        result := n
    end
    else
    begin
        var n1: Integer
        var n2: Integer
        n1 := n - 1
        n2 := n - 2

        result := fibonacci(n1) + fibonacci(n2)
    end

    return result
end

procedure main()
begin
    var n: Integer
    n := fibonacci(10)
end
